    <script>
        // ========== ZOOM AND PAN FUNCTIONALITY ==========
        let zoomLevel = 100;
        let scrollLeft = 0;
        let scrollTop = 0;
        const minZoom = 50;
        const maxZoom = 300;
        const zoomStep = 10;
        const panStep = 50;

        const gridViewport = document.getElementById('grid-viewport');
        const gridContainer = document.getElementById('grid-container');
        const zoomPercentSpan = document.getElementById('zoom-percent');

        function updateZoom() {
            const scale = zoomLevel / 100;
            gridContainer.style.transform = `scale(${scale})`;
            gridContainer.style.transformOrigin = 'top left';
            zoomPercentSpan.textContent = zoomLevel;
        }

        function applyScroll() {
            gridViewport.scrollLeft = scrollLeft;
            gridViewport.scrollTop = scrollTop;
        }

        document.getElementById('zoom-in').addEventListener('click', () => {
            if (zoomLevel < maxZoom) {
                zoomLevel += zoomStep;
                updateZoom();
            }
        });

        document.getElementById('zoom-out').addEventListener('click', () => {
            if (zoomLevel > minZoom) {
                zoomLevel -= zoomStep;
                updateZoom();
            }
        });

        document.getElementById('pan-left').addEventListener('click', () => {
            scrollLeft = Math.max(0, scrollLeft - panStep);
            applyScroll();
        });

        document.getElementById('pan-right').addEventListener('click', () => {
            scrollLeft = Math.min(gridContainer.scrollWidth, scrollLeft + panStep);
            applyScroll();
        });

        document.getElementById('pan-up').addEventListener('click', () => {
            scrollTop = Math.max(0, scrollTop - panStep);
            applyScroll();
        });

        document.getElementById('pan-down').addEventListener('click', () => {
            scrollTop = Math.min(gridContainer.scrollHeight, scrollTop + panStep);
            applyScroll();
        });

        document.getElementById('reset').addEventListener('click', () => {
            zoomLevel = 100;
            scrollLeft = 0;
            scrollTop = 0;
            updateZoom();
            applyScroll();
        });

        // Wheel zoom
        gridViewport.addEventListener('wheel', (e) => {
            e.preventDefault();
            if (e.deltaY > 0) {
                if (zoomLevel > minZoom) {
                    zoomLevel -= zoomStep;
                }
            } else {
                if (zoomLevel < maxZoom) {
                    zoomLevel += zoomStep;
                }
            }
            updateZoom();
        });

        // Keyboard navigation
        document.addEventListener('keydown', (e) => {
            if (document.activeElement === gridViewport || document.activeElement === document.body) {
                switch (e.key) {
                    case 'ArrowUp':
                        e.preventDefault();
                        scrollTop = Math.max(0, scrollTop - 20);
                        applyScroll();
                        break;
                    case 'ArrowDown':
                        e.preventDefault();
                        scrollTop = Math.min(gridContainer.scrollHeight, scrollTop + 20);
                        applyScroll();
                        break;
                    case 'ArrowLeft':
                        e.preventDefault();
                        scrollLeft = Math.max(0, scrollLeft - 20);
                        applyScroll();
                        break;
                    case 'ArrowRight':
                        e.preventDefault();
                        scrollLeft = Math.min(gridContainer.scrollWidth, scrollLeft + 20);
                        applyScroll();
                        break;
                    case '+':
                    case '=':
                        e.preventDefault();
                        if (zoomLevel < maxZoom) {
                            zoomLevel += zoomStep;
                            updateZoom();
                        }
                        break;
                    case '-':
                        e.preventDefault();
                        if (zoomLevel > minZoom) {
                            zoomLevel -= zoomStep;
                            updateZoom();
                        }
                        break;
                }
            }
        });

        // ========== STATISTICS CALCULATION ==========
        function updateStatistics() {
            const cells = document.querySelectorAll('.cell');
            const redCells = document.querySelectorAll('.cell.red');
            const totalCells = cells.length;
            const redCount = redCells.length;
            const whiteCells = totalCells - redCount;
            const redPercentage = totalCells > 0 ? ((redCount / totalCells) * 100).toFixed(1) : 0;

            document.getElementById('total-cells').textContent = totalCells.toLocaleString();
            document.getElementById('red-cells').textContent = redCount.toLocaleString();
            document.getElementById('white-cells').textContent = whiteCells.toLocaleString();
            document.getElementById('red-percent').textContent = redPercentage + '%';
        }

        // ========== DOWNLOAD FUNCTIONALITY ==========
        document.getElementById('download-btn').addEventListener('click', async function() {
            const btn = this;
            const loading = document.getElementById('loading');
            
            btn.disabled = true;
            btn.textContent = '‚è≥ En cours...';
            loading.classList.add('active');

            try {
                // Reset zoom and scroll for clean screenshot
                const originalZoom = zoomLevel;
                const originalScrollLeft = scrollLeft;
                const originalScrollTop = scrollTop;

                zoomLevel = 100;
                scrollLeft = 0;
                scrollTop = 0;
                updateZoom();
                applyScroll();

                setTimeout(async () => {
                    const container = document.querySelector('.main-container');
                    const canvas = await html2canvas(container, {
                        scale: 2,
                        backgroundColor: '#ffffff'
                    });
                    
                    const link = document.createElement('a');
                    link.download = 'pyramide_sierpinski.png';
                    link.href = canvas.toDataURL();
                    link.click();

                    // Restore zoom and scroll
                    zoomLevel = originalZoom;
                    scrollLeft = originalScrollLeft;
                    scrollTop = originalScrollTop;
                    updateZoom();
                    applyScroll();

                    btn.textContent = '‚úì T√©l√©charg√©!';
                    setTimeout(() => {
                        btn.disabled = false;
                        btn.textContent = 'üì• T√©l√©charger';
                        loading.classList.remove('active');
                    }, 1500);
                }, 100);
            } catch (error) {
                console.error('Erreur:', error);
                btn.textContent = '‚ùå Erreur';
                btn.disabled = false;
                loading.classList.remove('active');
                setTimeout(() => {
                    btn.textContent = 'üì• T√©l√©charger';
                }, 2000);
            }
        });

        // ========== LINE SEARCH FUNCTIONALITY ==========
        const zippedDataArray = [
            <% zippedData.forEach((item, idx) => { %>
                { line: <%= idx + 1 %>, redCount: <%= item.redCount %> },
            <% }); %>
        ];

        // Count bits in binary representation
        function countBits(n) {
            let count = 0;
            while (n) {
                count += n & 1;
                n >>= 1;
            }
            return count;
        }

        document.getElementById('search-line').addEventListener('click', () => {
            const lineNum = parseInt(document.getElementById('line-number').value);
            const resultDiv = document.getElementById('search-result');
            const formulaDiv = document.getElementById('formula-display');

            if (isNaN(lineNum) || lineNum < 1 || lineNum > zippedDataArray.length) {
                resultDiv.textContent = '‚ùå Ligne invalide';
                resultDiv.classList.add('not-found');
                resultDiv.style.display = 'block';
                formulaDiv.innerHTML = '';
                return;
            }

            const lineData = zippedDataArray[lineNum - 1];
            resultDiv.textContent = `Ligne ${lineNum}: ${lineData.redCount} üî¥`;
            resultDiv.classList.remove('not-found');
            resultDiv.style.display = 'block';

            // Calculate formula
            const n = lineNum - 1; // 0-indexed
            const bitCount = countBits(n);
            const formulaResult = Math.pow(2, bitCount);
            const binaryN = n.toString(2);
            const totalCells = lineData.redCount + (zippedDataArray[lineNum - 1].redCount === lineData.redCount ? 0 : 0);
            const gridWidth = <%= gridWidth %>;

            formulaDiv.innerHTML = `
                <div class="formula-box">
                    <strong>üìê Formule:</strong><br/>
                    Ligne ${lineNum} (index ${n}) en binaire = <code>${binaryN}</code><br/>
                    Nombre de 1 en binaire = <code>${bitCount}</code><br/>
                    <strong>Carr√©s rouges = 2<sup>${bitCount}</sup> = ${formulaResult}</strong><br/>
                    <br/>
                    <strong>Total carr√©s:</strong> ${gridWidth}<br/>
                    <strong>Carr√©s blancs:</strong> ${gridWidth - lineData.redCount}
                </div>
            `;
        });

        document.getElementById('line-number').addEventListener('keypress', (e) => {
            if (e.key === 'Enter') {
                document.getElementById('search-line').click();
            }
        });

        // Initialize
        updateStatistics();
        updateZoom();
    </script>
